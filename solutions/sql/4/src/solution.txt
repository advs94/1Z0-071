4. SELECT employee_id,
          last_name,
          salary,
          ROUND(salary * 1.15) AS "New Salary",
          ROUND(salary * 1.15) - salary AS "Increase"
     FROM employees;
5. SELECT INITCAP(last_name) AS "Name",
          LENGTH(last_name) AS "Length"
     FROM employees
    WHERE last_name LIKE 'A%'
       OR last_name LIKE 'M%'
    ORDER BY last_name ASC;
   a. SELECT INITCAP(last_name) AS "Name",
             LENGTH(last_name) AS "Length"
        FROM employees
       WHERE last_name LIKE '&start_letter%'
       ORDER BY last_name ASC;
   b. SELECT INITCAP(last_name) AS "Name",
             LENGTH(last_name) AS "Length"
        FROM employees
       WHERE last_name LIKE UPPER('&start_letter%')
       ORDER BY last_name ASC;
6. SELECT last_name,
          ROUND(MONTHS_BETWEEN(SYSDATE, hire_date)) AS months_worked
     FROM employees
    ORDER BY months_worked;
7. SELECT last_name,
          LPAD(salary, 15, '$') AS salary
     FROM employees;
8. SELECT last_name,
          RPAD(' ', (salary/1000) + 1, '*') AS salaries_in_asterisk
     FROM employees
    ORDER BY salary DESC;
9. SELECT last_name,
          TRUNC((SYSDATE - hire_date) / 7) AS tenure
     FROM employees
    WHERE department_id = 90
    ORDER BY tenure DESC;
